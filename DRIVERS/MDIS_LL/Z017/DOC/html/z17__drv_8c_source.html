477 static int32 Z17_SetStat(
478  LL_HANDLE *llHdl,
479  int32 code,
480  int32 ch,
481  INT32_OR_64 value32_or_64
482 )
483 {
484  int32 value = (int32)value32_or_64; /* 32bit value */
485 #ifndef Z127_NOIRQ
486  INT32_OR_64 valueP = value32_or_64; /* stores 32/64bit pointer */
487  M_SG_BLOCK *blk = (M_SG_BLOCK*)valueP; /* stores block struct pointer */
488 #endif
489  MACCESS ma = llHdl->ma;
490  int32 error = ERR_SUCCESS;
491 
492  DBGWRT_1((DBH, "LL - Z17_SetStat: ch=%d code=0x%04x value=0x%x\n",
493  ch, code, value));
494 
495  switch (code) {
496  /*--------------------------+
497  | debug level |
498  +--------------------------*/
499  case M_LL_DEBUG_LEVEL:
500  llHdl->dbgLevel = value;
501  break;
502  /*--------------------------+
503  | enable interrupts |
504  +--------------------------*/
505  case M_MK_IRQ_ENABLE:
506  break;
507  /*--------------------------+
508  | set irq counter |
509  +--------------------------*/
510  case M_LL_IRQ_COUNT:
511 #ifndef Z127_NOIRQ
512  llHdl->irqCount = value;
513 #endif
514  break;
515  /*--------------------------+
516  | channel direction |
517  +--------------------------*/
518  case M_LL_CH_DIR:
519  if (value != M_CH_INOUT) {
520  error = ERR_LL_ILL_DIR;
521  }
522  break;
523  /*--------------------------+
524  | set IO ports |
525  +--------------------------*/
526  case Z17_SET_PORTS:
527  MWRITE_D32(ma, ARWEN_GPIO, MREAD_D32(ma, ARWEN_GPIO) | value);
528  /* remove bit from toggle mode */
529  if (llHdl->togBitMask) {
530  llHdl->togBitMask &= ~value;
531  /* stop alarm if last bit is set */
532  if (!llHdl->togBitMask)
533  error = OSS_AlarmClear(llHdl->osHdl, llHdl->alarmHdl);
534  }
535  break;
536  /*--------------------------+
537  | clear IO ports |
538  +--------------------------*/
539  case Z17_CLR_PORTS:
540  MWRITE_D32(ma, ARWEN_GPIO, MREAD_D32(ma, ARWEN_GPIO) & ~value);
541  /* remove bit from toggle mode */
542  if (llHdl->togBitMask) {
543  llHdl->togBitMask &= ~value;
544  /* stop alarm if last bit is cleared */
545  if (!llHdl->togBitMask)
546  error = OSS_AlarmClear(llHdl->osHdl, llHdl->alarmHdl);
547  }
548  break;
549  /*--------------------------+
550  | port direction |
551  +--------------------------*/
552  case Z17_DIRECTION:
553  MWRITE_D32(ma, ARWEN_DDR, value);
554  break;
555 #ifndef Z127_NOIRQ
556  /*--------------------------+
557  | edge for interrupt |
558  +--------------------------*/
559  case Z17_IRQ_SENSE:
560 #ifndef Z17_MODEL_Z127
561  /* Z34/Z37 - 8 GPIOs */
562  MWRITE_D32(ma, ARWEN_02_IER1, value & 0xff);
563  MWRITE_D32(ma, ARWEN_02_IER2, (value >> 8) & 0xff);
564  break;
565 #else
566  /* Z127: GPIOs 0..15 - two bits per GPIO */
567  MWRITE_D32(ma, ARWEN_02_IER1, value);
568  break;
569  /* Z127: GPIOs 16..31 - two bits per GPIO */
570  case Z17_IRQ_SENSE_16TO31:
571  MWRITE_D32(ma, ARWEN_02_IER2, value);
572  break;
573 #endif /* Z17_MODEL_Z127 */
574 #endif /* Z127_NOIRQ */
575  /*--------------------------+
576  | open drain |
577  +--------------------------*/
578  case Z17_OPEN_DRAIN:
579  MWRITE_D32(ma, ARWEN_02_ODER, value);
580  break;
581  /*--------------------------+
582  | debouncer |
583  +--------------------------*/
584  case Z17_DEBOUNCE:
585  MWRITE_D32(ma, ARWEN_02_DBER, value);
586  break;
587 
588 #ifdef Z127_INFO_PTR
589  case Z17_BLK_DEBOUNCE_TIME:
590  {
591  int8 port;
592  Z17_BLK_DEBTIME *dbt = (Z17_BLK_DEBTIME*)blk->data;
593 
594  /* prevent access to not implemented DBCR registers */
595  if( llHdl->vdt == 0 ){
596  DBGWRT_ERR((DBH, "*** LL - Z17_SetStat(Z17_BLK_DEBOUNCE_TIME): "
597  " requires at least 16Z127 variant 1 IP core\n"));
598  return ERR_LL_ILL_FUNC;
599  }
600 
601  for( port=0; port<32; port++ ){
602  if( (dbt->portMask & (1<<port)) ){
603  MWRITE_D32(ma, ARWEN_Z127V01_DBCR(port), dbt->timeUs/50);
604  }
605  }
606 
607  break;
608  }
609 #endif
610 
611 #ifndef Z127_NOIRQ
612  /*--------------------------+
613  | register signal |
614  +--------------------------*/
615  case Z17_SET_SIGNAL:
616  /* signal already installed ? */
617  if (llHdl->portChangeSig) {
618  error = ERR_OSS_SIG_SET;
619  break;
620  }
621  error = OSS_SigCreate(OSH, value, &llHdl->portChangeSig);
622  break;
623  /*--------------------------+
624  | unregister signal |
625  +--------------------------*/
626  case Z17_CLR_SIGNAL:
627  /* signal already installed ? */
628  if (llHdl->portChangeSig == NULL) {
629  error = ERR_OSS_SIG_CLR;
630  break;
631  }
632  error = OSS_SigRemove(OSH, &llHdl->portChangeSig);
633  break;
634  /*--------------------------+
635  | init IRQ latency test |
636  +--------------------------*/
637  case Z17_BLK_IRQLAT_START:
638  {
639  int32 value;
640  Z17_BLK_IRQLAT *irqlat = (Z17_BLK_IRQLAT*)blk->data;
641 
642  /* clear out values */
643  irqlat->startTick = 0;
644  irqlat->stopTick = 0;
645  irqlat->irqsRcved = 0;
646 
647  /* init */
648  llHdl->irqTest = 1;
649  llHdl->irqs2fire = irqlat->irqs2fire-1;
650  llHdl->irqCount = 0;
651  llHdl->stopTick = 0;
652 
653  /* compute and save output bit */
654  llHdl->outBit = 1 << irqlat->outPort;
655 
656  /* disable all interrupts */
657  MWRITE_D32(ma, ARWEN_02_IER1, 0);
658  MWRITE_D32(ma, ARWEN_02_IER2, 0);
659 
660  /* program all ports as inputs */
661  MWRITE_D32(ma, ARWEN_DDR, 0);
662 
663  /* determine actual state of output port */
664  value = MREAD_D32(ma, ARWEN_02_PSR);
665  llHdl->outLast = (value & llHdl->outBit) ? 1 : 0;
666 
667  /* preset output port value */
668  MWRITE_D32(ma, ARWEN_GPIO, value);
669 
670  /* enable IRQ for rising+falling edge of input port */
671 /* Z34/Z37 - 8 GPIOs */
672 #ifndef Z17_MODEL_Z127
673  if (irqlat->inPort < 4) /* port 3..0 */
674  MWRITE_D32(ma, ARWEN_02_IER1, 3 << (irqlat->inPort * 2));
675  else /* port 8..4 */
676  MWRITE_D32(ma, ARWEN_02_IER2, 3 << ((irqlat->inPort-4) * 2));
677 /* Z127 - 32 GPIOs */
678 #else
679  if (irqlat->inPort < 16) /* port 15..0 */
680  MWRITE_D32(ma, ARWEN_02_IER1, 3 << (irqlat->inPort * 2));
681  else /* port 32..16 */
682  MWRITE_D32(ma, ARWEN_02_IER2, 3 << ((irqlat->inPort-16) * 2));
683 #endif /* Z17_MODEL_Z127 */
684  /* config port as output (default is input) */
685  MWRITE_D32(ma, ARWEN_DDR, llHdl->outBit);
686 
687  /* set output (this fires the first IRQ) */
688  llHdl->outLast = !llHdl->outLast;
689  DBGWRT_3((DBH, "set output=0x%x, irqs2fire=%d\n",
690  llHdl->outLast, llHdl->irqs2fire));
691  llHdl->startTick = OSS_TickGet(OSH);
692  MWRITE_D32(ma, ARWEN_GPIO, llHdl->outLast ? llHdl->outBit : 0);
693  break;
694  }
695 #endif /* Z127_NOIRQ */
696  /*--------------------------+
697  | Toggle mode |
698  +--------------------------*/
699  case Z17_TOG_PORTS:
700  {
701  u_int32 realMsec;
702 
703  /* start timer initially */
704  if (!llHdl->togBitMask) {
705  llHdl->togBitMask = value;
706  error = OSS_AlarmSet(llHdl->osHdl, llHdl->alarmHdl,
